   if ( x >= y )
      max = x;

Example translating an if-else statement
  

main:
        movw    r0, #:lower16:x
        movt    r0, #:upper16:x         // r0 = addr(x)
        ldr     r0, [r0]                // r0 = x

        movw    r1, #:lower16:y
        movt    r1, #:upper16:y         // r1 = addr(y)
        ldr     r1, [r1]                // r1 = y

        cmp     r0, r1                  // compare x ? y
        blt     else                    // if ( x < y ) Branch to "else" 

        // max = x
        movw    r0, #:lower16:x
        movt    r0, #:upper16:x         // r0 = addr(x)
        ldr     r0, [r0]                // r0 = x

        movw    r2, #:lower16:max
        movt    r2, #:upper16:max       // r2 = addr(max)
        str     r0, [r2]                // max = r0 (= x)

        b       ifEnd                   // Skip over the else part !!!

else:   // max = y
        movw    r0, #:lower16:y
        movt    r0, #:upper16:y         // r0 = addr(y)
        ldr     r0, [r0]                // r0 = y

        movw    r2, #:lower16:max
        movt    r2, #:upper16:max       // r2 = addr(max)
        str     r0, [r2]                // max = r0 (= y)

ifEnd:

/* --------------------------------------------------
   Begin of the permanent program variables
   -------------------------------------------------- */
        .data

x:      .4byte  7      // Try a value > y
y:      .4byte  10
max:    .kip    4
